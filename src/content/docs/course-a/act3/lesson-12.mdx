"Raspberry Pi Pico WH에서 Grove Shield를 사용해요. I2C 포트에 DHT20 온습도 센서와 SSD1306 OLED(128x64)가 연결되어 있어요. A0 포트에 소리 센서가 있어요. OLED에 실시간 대시보드를 만들어주세요. 첫 줄에 온도와 습도를 큰 글씨로, 두 번째 줄에 소음 레벨 막대그래프를, 세 번째 줄에 쾌적도 상태(좋음/보통/나쁨)를 표시해주세요. 1초마다 갱신하고, print()로도 데이터를 출력해주세요. MicroPython으로 작성해주세요."

### OLED 종합 대시보드 (MicroPython)

```python
from machine import I2C, Pin, ADC
import ssd1306
import time

# I2C 설정 (OLED + DHT20 공유)
i2c = I2C(0, sda=Pin(8), scl=Pin(9))
time.sleep(0.1)

# OLED 디스플레이 초기화
oled = ssd1306.SSD1306_I2C(128, 64, i2c)

# DHT20 센서 주소
dht_addr = 0x38

# 소리 센서 (ADC)
sound = ADC(Pin(26))  # A0 포트

def read_dht20():
    """DHT20에서 온도와 습도를 읽습니다"""
    i2c.writeto(dht_addr, bytes([0xAC, 0x33, 0x00]))
    time.sleep(0.08)
    data = i2c.readfrom(dht_addr, 7)
    humidity = ((data[1] << 12) | (data[2] << 4) | (data[3] >> 4)) / 1048576 * 100
    temperature = (((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5]) / 1048576 * 200 - 50
    return round(temperature, 1), round(humidity, 1)

def read_sound_level():
    """소리 센서의 값을 0~100 레벨로 변환합니다"""
    raw = sound.read_u16()
    level = min(100, int(raw / 65535 * 100))
    return level

def draw_bar(oled, x, y, width, height, value, max_val=100):
    """OLED에 막대그래프를 그립니다"""
    oled.rect(x, y, width, height, 1)  # 테두리
    fill_width = int((value / max_val) * (width - 2))
    if fill_width > 0:
        oled.fill_rect(x + 1, y + 1, fill_width, height - 2, 1)

def get_comfort_status(temp, humi):
    """온습도에 따른 쾌적도 상태를 반환합니다"""
    if 20 <= temp <= 26 and 40 <= humi <= 60:
        return "GOOD"
    elif 18 <= temp <= 28 and 30 <= humi <= 70:
        return "OK"
    else:
        return "BAD"

# 메인 루프: 실시간 대시보드
print("=== 실시간 대시보드 시작 ===")

while True:
    try:
        # 센서 데이터 읽기
        temp, humi = read_dht20()
        noise = read_sound_level()
        status = get_comfort_status(temp, humi)

        # OLED 화면 그리기
        oled.fill(0)  # 화면 지우기

        # 1행: 온도와 습도 (큰 정보)
        oled.text(f"{temp}C", 0, 0, 1)
        oled.text(f"{humi}%", 72, 0, 1)

        # 2행: 구분선
        oled.hline(0, 12, 128, 1)

        # 3행: 소음 레벨 막대그래프
        oled.text("Noise", 0, 16, 1)
        draw_bar(oled, 48, 16, 76, 10, noise)

        # 4행: 습도 막대그래프
        oled.text("Humid", 0, 30, 1)
        draw_bar(oled, 48, 30, 76, 10, humi)

        # 5행: 쾌적도 상태
        oled.text(f"Status: {status}", 0, 46, 1)

        # 6행: 업데이트 표시
        oled.text("* LIVE *", 40, 56, 1)

        oled.show()  # 화면 갱신

        # Serial 출력 (PC 대시보드용)
        print(f"{temp},{humi},{noise},{status}")

    except Exception as e:
        oled.fill(0)
        oled.text("ERROR", 0, 0, 1)
        oled.text(str(e)[:16], 0, 12, 1)
        oled.show()

    time.sleep(1)
```

### 줄별 해설

| 줄 | 코드 | 설명 |
|----|------|------|
| 7 | `ssd1306.SSD1306_I2C(128, 64, i2c)` | 128x64 픽셀 OLED를 I2C로 초기화해요. |
| 32 | `oled.rect(x, y, w, h, 1)` | 사각형 테두리를 그려요. 마지막 `1`은 흰색이에요. |
| 33 | `oled.fill_rect(...)` | 채워진 사각형을 그려요. 막대그래프의 채움 부분이에요. |

### PC 실시간 대시보드 (Python + matplotlib)

PC에서 실행하는 코드예요. Pico가 serial로 보내는 데이터를 받아 실시간 그래프를 그려요.

```python
# PC에서 실행 (Thonny가 아닌 터미널에서)
# pip install pyserial matplotlib 필요

import serial
import matplotlib.pyplot as plt
from collections import deque
import time

# Serial 포트 설정 (OS에 따라 다름)
# Windows: "COM3", "COM4" 등
# Mac: "/dev/tty.usbmodem..."
# Linux: "/dev/ttyACM0"
PORT = "COM3"  # 여러분의 포트로 수정하세요
BAUD = 115200

# 최근 60개 데이터만 유지 (60초분)
temps = deque(maxlen=60)
humis = deque(maxlen=60)
noises = deque(maxlen=60)

# 그래프 설정
plt.ion()  # 인터랙티브 모드
fig, (ax1, ax2, ax3) = plt.subplots(3, 1, figsize=(10, 8))
fig.suptitle("Real-time Sensor Dashboard", fontsize=14)

try:
    ser = serial.Serial(PORT, BAUD, timeout=1)
    print(f"연결됨: {PORT}")

    while True:
        line = ser.readline().decode().strip()
        if not line:
            continue

        try:
            parts = line.split(",")
            temp = float(parts[0])
            humi = float(parts[1])
            noise = float(parts[2])
            status = parts[3] if len(parts) > 3 else ""

            temps.append(temp)
            humis.append(humi)
            noises.append(noise)
        except (ValueError, IndexError):
            continue

        # 그래프 갱신
        for ax in (ax1, ax2, ax3):
            ax.clear()

        ax1.plot(list(temps), color="tomato", linewidth=2)
        ax1.set_ylabel("Temp (C)")
        ax1.set_ylim(15, 35)
        ax1.grid(True, alpha=0.3)

        ax2.plot(list(humis), color="steelblue", linewidth=2)
        ax2.set_ylabel("Humidity (%)")
        ax2.set_ylim(20, 80)
        ax2.grid(True, alpha=0.3)

        ax3.bar(range(len(noises)), list(noises), color="orange")
        ax3.set_ylabel("Noise Level")
        ax3.set_ylim(0, 100)
        ax3.set_xlabel("Time (seconds)")

        plt.tight_layout()
        plt.pause(0.1)

except serial.SerialException as e:
    print(f"포트 연결 실패: {e}")
    print("Thonny를 종료한 뒤 다시 시도하세요.")
except KeyboardInterrupt:
    print("대시보드 종료")
finally:
    plt.ioff()
    if 'ser' in dir():
        ser.close()
```

:::caution[Thonny와 serial 충돌]
Thonny가 Pico에 연결되어 있으면, PC Python에서 serial 포트를 열 수 없어요. **Thonny를 완전히 종료**한 뒤에 이 코드를 실행하세요. 코드를 `main.py`로 Pico에 저장해두면, Thonny 없이도 Pico가 자동으로 실행됩니다.
:::

:::note[deque란?]
`deque(maxlen=60)`은 최대 60개까지만 데이터를 저장하는 자료구조예요. 새 데이터가 들어오면 가장 오래된 데이터가 자동으로 빠져요. 실시간 대시보드에서 최근 데이터만 보여줄 때 유용해요.
:::

## 잠깐, 방금 무슨 일이 일어난 걸까요?

여러분이 방금 한 일을 다시 생각해보세요.

작은 OLED 화면에 **실시간으로 변하는 숫자와 그래프**가 나타나고 있어요. 동시에 PC 화면에서도 **같은 데이터가 실시간 차트**로 그려지고 있어요.

이것은 단순한 "센서 읽기"가 아니에요. 여러분은 방금 **IoT(사물인터넷) 시스템의 핵심 구조**를 만든 거예요.

- **센서 계층**: 물리 세계의 데이터를 수집
- **엣지 디바이스**: Pico에서 데이터를 처리하고 OLED에 표시
- **통신 계층**: serial로 PC에 데이터 전송
- **클라우드/대시보드**: PC에서 실시간 시각화

스마트홈의 온도 조절기, 공장의 생산 모니터링 시스템, 병원의 환자 모니터 -- 모두 이 구조의 확장이에요.
